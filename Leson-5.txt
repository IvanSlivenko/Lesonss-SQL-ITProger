Вибрати всі данні з таблиці people
-------------------------------------------
SELECT * FROM `people`
-------------------------------------------





Вибрати поля name та bio з таблиці  people
------------------------------------------
SELECT `name`, `bio` FROM `people`;
------------------------------------------



Вибрати всі стрічки з полями id та name де id>2 і id < 4
----------------------------------------------------------
SELECT `id`,`name` FROM `people` WHERE id > 2 AND id < 4;
----------------------------------------------------------



--------------------------------------------------------
SELECT * FROM `people` WHERE id <> 5
AND `name`='Іван'
; 
--------------------------------------------------------

--------------------------------------------------------
SELECT * FROM `people` WHERE id <> 5
AND name='Іван'
AND bio IS NULL 
;
-------------------------------------------------------- 

--------------------------------------------------------
SELECT * FROM `people` WHERE id <> 5 AND name='Іван' OR id = 4;
--------------------------------------------------------



Вибрати всі унікальні поля name з people
--------------------------------------------------------
SELECT DISTINCT name FROM people;
--------------------------------------------------------




Вибрати перші два записи
-------------------------------------------------------
SELECT * FROM `people` LIMIT 2;
-------------------------------------------------------

Вибрати     три     записи, пропускаючи перед цим      два     записи
----------------------------------------------------------------------
SELECT * FROM `people` LIMIT 2, 3;

ORDER BY id  додає сортування по полю id за сростанням
SELECT * FROM `people` ORDER BY id LIMIT 2, 3;
---------------------------------------------------------------------

--------------------------------------------------------------------
ORDER BY id DESC додає сортування по полю id за спаданням
SELECT * FROM `people` ORDER BY id DESC LIMIT 2, 3;
