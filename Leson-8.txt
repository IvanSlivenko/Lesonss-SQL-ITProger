--------------------------------------------------------
Вибираємо мінімальне значення

SELECT MIN(price) FROM shop;
--------------------------------------------------------


---------------------------------------------------------
Вибираємо максимальне значення

SELECT MAX(price) AS 'Максимальна ціна' FROM shop;
----------------------------------------------------------


-----------------------------------------------------------
Вибираємо середнє значення

SELECT AVG(price) AS 'Середня ціна' FROM shop;
-----------------------------------------------------------


-----------------------------------------------------------
Вибириємо сумму всіх елементів

SELECT SUM(price) AS 'Сумма' FROM shop;
-----------------------------------------------------------


-----------------------------------------------------------
Переводимо вибрані даанні в верхній регістр
SELECT UCASE(title)  FROM shop;
-----------------------------------------------------------


-----------------------------------------------------------
Переводимо вибрані даанні в нижній регістр
SELECT LCASE(title) FROM shop;

GROUP BY price
HAVING COUNT(price)>1
;
